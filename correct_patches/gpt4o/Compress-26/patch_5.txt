public static long skip(InputStream input, long numToSkip) throws IOException {
    long available = numToSkip;
    byte[] buffer = new byte[1]; // Using a buffer to read and discard bytes
    while (numToSkip > 0) {
        long skipped = input.skip(numToSkip);
        if (skipped == 0) {
            // Fallback to reading if skip doesn't advance the stream
            int read = input.read(buffer, 0, 1);
            if (read == -1) { // If we reach the end of the stream
                break;
            }
            skipped = 1; // 1 byte was effectively skipped by reading
        }
        numToSkip -= skipped;
    }
    return available - numToSkip;
}
