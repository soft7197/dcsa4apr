static boolean isSimpleNumber(String s) {
  if (s == null || s.isEmpty()) {
    return false;
  }
  int len = s.length();
  for (int index = 0; index < len; index++) {
    char c = s.charAt(index);
    if (!Character.isDigit(c)) {
      return false;
    }
  }
  // A valid simple number should not start with '0' unless it's '0'
  return !(len > 1 && s.charAt(0) == '0');
}
