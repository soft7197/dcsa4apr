protected double getInitialDomain(double p) {
    double ret;
    double d = getDenominatorDegreesOfFreedom();
    if (d > 2.0) {
        ret = d / (d - 2.0);
    } else if (d > 0) {
        ret = 1.0; // reasonable fallback for positive but small d
    } else {
        ret = Double.POSITIVE_INFINITY; // fallback for non-positive d to highlight the mathematical restriction
    }
    return ret;
}
