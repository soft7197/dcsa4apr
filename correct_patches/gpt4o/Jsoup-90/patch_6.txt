private static boolean looksLikeUtf8(byte[] input) {
    int i = 0;
    // BOM:
    if (input.length >= 3 && (input[0] & 0xFF) == 0xEF
        && (input[1] & 0xFF) == 0xBB && (input[2] & 0xFF) == 0xBF) {
        i = 3;
    }

    while (i < input.length) {
        int o = input[i];
        if ((o & 0x80) == 0) {
            i++; // ASCII
            continue;
        }

        int bytesToProcess;
        // UTF-8 leading bytes:
        if ((o & 0xE0) == 0xC0) {
            bytesToProcess = 1;
        } else if ((o & 0xF0) == 0xE0) {
            bytesToProcess = 2;
        } else if ((o & 0xF8) == 0xF0) {
            bytesToProcess = 3;
        } else {
            return false;
        }

        // Ensure there are enough bytes to form a valid UTF-8 sequence
        if (i + bytesToProcess >= input.length) {
            return false;
        }

        while (bytesToProcess > 0) {
            i++;
            o = input[i];
            if ((o & 0xC0) != 0x80) {
                return false; // invalid UTF-8
            }
            bytesToProcess--;
        }

        i++;
    }
    return true;
}
