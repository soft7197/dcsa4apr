int nextIndexOf(CharSequence seq) {
    if (seq == null || seq.length() == 0 || seq.length() > (length - pos)) {
        return -1;
    }
    char startChar = seq.charAt(0);
    int seqLen = seq.length();
    int lastPossibleIndex = length - seqLen;

    for (int offset = pos; offset <= lastPossibleIndex; offset++) {
        if (startChar != input[offset]) {
            while (++offset <= lastPossibleIndex && startChar != input[offset]);
        }

        if (offset <= lastPossibleIndex) {
            int i = offset + 1;
            int j = 1;
            while (j < seqLen && seq.charAt(j) == input[i]) {
                i++;
                j++;
            }
            if (j == seqLen) {
                return offset - pos;
            }
        }
    }
    return -1;
}
