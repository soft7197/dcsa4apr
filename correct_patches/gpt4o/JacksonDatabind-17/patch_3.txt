public boolean useForType(JavaType t) {
    while (t.isArrayType()) {
        t = t.getContentType();
    }
    // Check for TreeNode before applying _appliesFor logic entirely
    if (TreeNode.class.isAssignableFrom(t.getRawClass())) {
        return false;
    }
    switch (_appliesFor) {
    case NON_CONCRETE_AND_ARRAYS:
        // Since array elements are processed, same logic as OBJECT_AND_NON_CONCRETE
    case OBJECT_AND_NON_CONCRETE:
        return (t.getRawClass() == Object.class) || !t.isConcrete();
    case NON_FINAL:
        return !t.isFinal();
    default:
        return (t.getRawClass() == Object.class);
    }
}
