public double cumulativeProbability(double x) throws MathException {
    final double dev = x - mean;
    // Handle extreme deviations from the mean
    if (dev < -40 * standardDeviation) {
        return 0.0;
    } else if (dev > 40 * standardDeviation) {
        return 1.0;
    }
    try {
        // Calculating the cumulative probability using the error function
        return 0.5 * (1.0 + Erf.erf(dev / (standardDeviation * FastMath.sqrt(2.0))));
    } catch (MaxIterationsExceededException ex) {
        // Distinguish between extreme range and convergence failure
        throw new MathException("Failed to converge while computing cumulative probability.", ex);
    }
}
