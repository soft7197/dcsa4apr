public OngoingInjecter filterCandidate(final Collection<Object> mocks, final Field field, final Object fieldInstance) {
    if (mocks.size() == 1) {
        final Object matchingMock = mocks.iterator().next();
        return new OngoingInjecter() {
            public boolean thenInject() {
                try {
                    // Verify if fieldInstance or matchingMock is null
                    if (fieldInstance == null || matchingMock == null) {
                        return false;
                    }
                    // Try property setter first
                    String setterName = "set" + Character.toUpperCase(field.getName().charAt(0)) + field.getName().substring(1);
                    java.lang.reflect.Method setter = null;
                    try {
                        setter = fieldInstance.getClass().getMethod(setterName, field.getType());
                        // Check if the parameter type of setter is compatible with matchingMock
                        if (setter != null && setter.getParameterTypes()[0].isAssignableFrom(matchingMock.getClass())) {
                            setter.invoke(fieldInstance, matchingMock);
                            return true;
                        }
                    } catch (NoSuchMethodException e) {
                        // setter not found or not compatible, continue
                    }
                    // Fall back to field setting
                    new FieldSetter(fieldInstance, field).set(matchingMock);
                    return true;
                } catch (Exception e) {
                    throw new MockitoException("Problems injecting dependency in " + field.getName(), e);
                }
            }
        };
    }

    return new OngoingInjecter() {
        public boolean thenInject() {
            return false;
        }
    };
}
