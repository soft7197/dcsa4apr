public List getValues(final Option option,
                      List defaultValues) {
    // initialize the return list
    List valueList = (List) values.get(option);

    // if no parsed values, use default values or provided default values
    if ((valueList == null) || valueList.isEmpty()) {
        valueList = defaultValues != null ? new ArrayList(defaultValues) : (List) this.defaultValues.get(option);
    } else {
        // add additional default values if necessary
        if (defaultValues != null && !defaultValues.isEmpty()) {
            valueList.addAll(defaultValues.subList(valueList.size(), defaultValues.size()));
        } else {
            List internalDefaults = (List) this.defaultValues.get(option);
            if (internalDefaults != null) {
                valueList.addAll(internalDefaults.subList(valueList.size(), internalDefaults.size()));
            }
        }
    }

    return valueList == null ? Collections.EMPTY_LIST : valueList;
}
