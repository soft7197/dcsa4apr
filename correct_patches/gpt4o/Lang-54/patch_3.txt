public static Locale toLocale(String str) {
    if (str == null) {
        return null;
    }
    int len = str.length();
    if (len < 2) {
        throw new IllegalArgumentException("Invalid locale format: " + str);
    }
    char ch0 = str.charAt(0);
    char ch1 = str.charAt(1);
    if (ch0 < 'a' || ch0 > 'z' || ch1 < 'a' || ch1 > 'z') {
        throw new IllegalArgumentException("Invalid locale format: " + str);
    }
    if (len == 2) {
        return new Locale(str, "");
    } else {
        if (str.charAt(2) != '_') {
            throw new IllegalArgumentException("Invalid locale format: " + str);
        }
        // Look ahead for the optional country and variant components
        int underscoreIndex = str.indexOf('_', 3);
        if (underscoreIndex == -1) {
            // No further underscore means country code only
            if (len == 5) {
                char ch3 = str.charAt(3);
                char ch4 = str.charAt(4);
                if (ch3 < 'A' || ch3 > 'Z' || ch4 < 'A' || ch4 > 'Z') {
                    throw new IllegalArgumentException("Invalid country format: " + str);
                }
                return new Locale(str.substring(0, 2), str.substring(3, 5));
            } else {
                throw new IllegalArgumentException("Invalid locale format: " + str);
            }
        } else {
            // Parsing possible country (if it exists) and variant parts
            int countryLength = underscoreIndex - 3;
            if (countryLength == 0) { // Missing country but has variant
                return new Locale(str.substring(0, 2), "", str.substring(underscoreIndex + 1));
            } else if (countryLength == 2) { // Country is available
                char ch3 = str.charAt(3);
                char ch4 = str.charAt(4);
                if (ch3 < 'A' || ch3 > 'Z' || ch4 < 'A' || ch4 > 'Z') {
                    throw new IllegalArgumentException("Invalid country format: " + str);
                }
                return new Locale(str.substring(0, 2), str.substring(3, 5), str.substring(underscoreIndex + 1));
            } else {
                throw new IllegalArgumentException("Invalid locale format: " + str);
            }
        }
    }
}
