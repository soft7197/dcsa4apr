public double cumulativeProbability(double x) throws MathException {
    double dev = x - mean;
    if (dev < -20 * standardDeviation) {
        return 0.0;
    } else if (dev > 20 * standardDeviation) {
        return 1.0;
    }
    return 0.5 * (1.0 + Erf.erf(dev / (standardDeviation * Math.sqrt(2.0))));
}
