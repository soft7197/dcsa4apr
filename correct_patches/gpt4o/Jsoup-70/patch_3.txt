static boolean preserveWhitespace(Node node) {
    int depth = 0;
    while (node != null && node instanceof Element && depth < 6) {
        Element el = (Element) node;
        if (el.tag.preserveWhitespace()) {
            return true;
        }
        node = el.parent();
        depth++;
    }
    return false;
}
