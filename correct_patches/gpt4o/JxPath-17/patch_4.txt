public JDOMAttributeIterator(NodePointer parent, QName name) {
    this.parent = parent;
    attributes = new ArrayList(); // Initialize the attributes list early
    if (parent.getNode() instanceof Element) {
        Element element = (Element) parent.getNode();
        String prefix = name.getPrefix();
        Namespace ns = null;
        if (prefix != null) {
            if (prefix.equals("xml")) {
                ns = Namespace.XML_NAMESPACE;
            } else {
                String uri = parent.getNamespaceResolver().getNamespaceURI(prefix);
                if (uri != null) {
                    ns = Namespace.getNamespace(prefix, uri);
                }
            }
            // If the namespace cannot be resolved, return an empty attribute list
            if (ns == null) {
                return;
            }
        }

        String lname = name.getName();
        if (!lname.equals("*")) {
            Attribute attr;
            if (ns != null) {
                attr = element.getAttribute(lname, ns);
            } else {
                attr = element.getAttribute(lname);
            }
            if (attr != null) {
                attributes.add(attr);
            }
        } else {
            List allAttributes = element.getAttributes();
            for (int i = 0; i < allAttributes.size(); i++) {
                Attribute attr = (Attribute) allAttributes.get(i);
                if (ns == null || attr.getNamespace().equals(ns)) {
                    attributes.add(attr);
                }
            }
        }
    }
}
