public Attribute(String key, String val, Attributes parent) {
    Validate.notNull(key);
    String trimmedKey = key.trim();
    Validate.notEmpty(trimmedKey); // Ensure trimmed key is not empty
    this.key = trimmedKey;
    this.val = val;
    this.parent = parent;
}
