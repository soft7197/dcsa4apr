public Attribute(String key, String val, Attributes parent) {
    Validate.notNull(key);
    key = key.trim();
    Validate.notEmpty(key); // trimming could potentially make empty, so we validate here
    this.key = key;
    this.val = val;
    this.parent = parent;
}
