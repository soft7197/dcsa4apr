public char[] getTextBuffer() {
    // Are we just using shared input buffer?
    if (_inputStart >= 0) {
        return (_inputBuffer != null) ? _inputBuffer : contentsAsArray();
    }
    if (_resultArray != null) return _resultArray;
    if (_resultString != null) {
        if (_resultArray == null) {
            _resultArray = _resultString.toCharArray();
        }
        return _resultArray;
    }
    // Ensure current segment is handled correctly
    if (!_hasSegments) {
        return (_currentSegment != null) ? _currentSegment : contentsAsArray(); // Returning contents ensures non-null return
    }
    // Create a non-segmented array if none exists and return
    return contentsAsArray();
}
