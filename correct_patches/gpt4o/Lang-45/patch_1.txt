public static String abbreviate(String str, int lower, int upper, String appendToEnd) {
    // initial parameter checks
    if (str == null) {
        return null;
    }
    if (str.length() == 0) {
        return StringUtils.EMPTY;
    }

    // if the lower value is greater than the length of the string, set to the length of the string
    lower = Math.min(lower, str.length());
    // if the upper value is -1 (i.e. no limit) or greater than length, set to the length
    upper = upper == -1 ? str.length() : Math.min(upper, str.length());
    // if upper is less than lower, raise it to lower
    upper = Math.max(upper, lower);

    StringBuffer result = new StringBuffer();
    int index = StringUtils.indexOf(str, " ", lower);
    if (index == -1 || index > upper) {
        result.append(str.substring(0, upper));
        if (upper != str.length()) {
            result.append(StringUtils.defaultString(appendToEnd));
        }
    } else {
        result.append(str.substring(0, index));
        result.append(StringUtils.defaultString(appendToEnd));
    }
    return result.toString();
}
