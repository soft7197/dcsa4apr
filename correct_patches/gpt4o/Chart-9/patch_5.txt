public TimeSeries createCopy(RegularTimePeriod start, RegularTimePeriod end) throws CloneNotSupportedException {

    if (start == null) {
        throw new IllegalArgumentException("Null 'start' argument.");
    }
    if (end == null) {
        throw new IllegalArgumentException("Null 'end' argument.");
    }
    if (start.compareTo(end) > 0) {
        throw new IllegalArgumentException(
                "Requires start on or before end.");
    }
    int startIndex = getIndex(start);
    if (startIndex < 0) { // start period not in series
        startIndex = -(startIndex + 1);
    }
    int endIndex = getIndex(end);
    if (endIndex < 0) { // end period not in series
        endIndex = -(endIndex + 1) - 1;
    }

    // Return an empty series if computed indices are out of valid bounds
    if (startIndex >= this.data.size() || startIndex > endIndex) {
        TimeSeries copy = (TimeSeries) super.clone();
        copy.data = new java.util.ArrayList();
        return copy;
    }
    
    // Adjust endIndex to ensure it does not go below the valid data range
    if (endIndex < startIndex) {
        endIndex = startIndex - 1;
    }

    return createCopy(startIndex, endIndex);
}
