public double chiSquare(double[] expected, long[] observed) throws IllegalArgumentException {
    if ((expected.length < 2) || (expected.length != observed.length)) {
        throw new IllegalArgumentException(
                "observed, expected array lengths incorrect");
    }
    if (!isPositive(expected) || !isNonNegative(observed)) {
        throw new IllegalArgumentException(
            "observed counts must be non-negative and expected counts must be positive");
    }
    double sumExpected = 0;
    long sumObserved = 0;
    for (int i = 0; i < observed.length; i++) {
        sumExpected += expected[i];
        sumObserved += observed[i];
    }
    double ratio = (double) sumObserved / sumExpected;
    double sumSq = 0.0d;
    double dev = 0.0d;
    // Apply the rescaling once and clone the expected array
    double[] rescaledExpected = expected.clone();
    for (int i = 0; i < rescaledExpected.length; i++) {
        rescaledExpected[i] *= ratio;
        if (rescaledExpected[i] == 0) {
            throw new IllegalArgumentException("Expected count cannot be zero after rescaling.");
        }
    }
    for (int i = 0; i < observed.length; i++) {
        dev = ((double) observed[i] - rescaledExpected[i]);
        sumSq += dev * dev / rescaledExpected[i];
    }
    return sumSq;
}
