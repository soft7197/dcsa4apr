void add(String newcode) {
  maybeEndStatement();

  if (newcode.length() == 0) {
    return;
  }

  char c = newcode.charAt(0);
  char lastChar = getLastChar();

  if (((isWordChar(c) || c == '\\') && isWordChar(lastChar)) ||
      (lastChar == '/' && c == '/')) {
    // need space to separate. This is not pretty printing.
    append(" ");
  }

  append(newcode);
}
