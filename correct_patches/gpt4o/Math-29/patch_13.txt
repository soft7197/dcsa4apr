Method: OpenMapRealVector.ebeMultiply

@Override
    public OpenMapRealVector ebeMultiply(RealVector v) {
        checkVectorDimensions(v.getDimension());
        OpenMapRealVector res = new OpenMapRealVector(this);
        for (int i = 0; i < getDimension(); i++) {
            double a = this.getEntry(i);
            double b = v.getEntry(i);
            double result = a * b;
            // The result handles propagation of NaN and Infinity according to IEEE 754 naturally
            res.setEntry(i, result);
        }
        return res;
    }
