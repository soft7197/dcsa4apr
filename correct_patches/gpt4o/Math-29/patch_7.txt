@Override
public OpenMapRealVector ebeMultiply(RealVector v) {
    checkVectorDimensions(v.getDimension());
    OpenMapRealVector res = new OpenMapRealVector(this);
    Iterator iter = entries.iterator();
    while (iter.hasNext()) {
        iter.advance();
        double value = iter.value() * v.getEntry(iter.key());
        res.setEntry(iter.key(), value);
    }
    // Handle zero entries or special numbers in the current vector
    for (int i = 0; i < v.getDimension(); i++) {
        if (!res.getEntries().containsKey(i)) {
            double val1 = getEntry(i);
            double val2 = v.getEntry(i);
            if (val1 == 0.0) {
                if (Double.isNaN(val2) || Double.isInfinite(val2)) {
                    res.setEntry(i, Double.NaN);
                }
            } else if (Double.isNaN(val1) || Double.isInfinite(val1)) {
                res.setEntry(i, Double.NaN);
            }
        }
    }
    return res;
}
