@Override
public OpenMapRealVector ebeMultiply(RealVector v) {
    checkVectorDimensions(v.getDimension());
    OpenMapRealVector res = new OpenMapRealVector(this);
    Iterator iter = entries.iterator();
    while (iter.hasNext()) {
        iter.advance();
        res.setEntry(iter.key(), iter.value() * v.getEntry(iter.key()));
    }
    // Handle special cases where multiplying `0d` in `this` vector with `v[i]` results in NaN or Infinity
    for (int i = 0; i < v.getDimension(); i++) {
        if (!res.entries.containsKey(i)) { // Check non-existing (i.e., zero entries)
            double valueV = v.getEntry(i);
            if (Double.isNaN(valueV) || Double.isInfinite(valueV)) {
                res.setEntry(i, Double.NaN);
            }
        }
    }
    return res;
}
