@Override
    public OpenMapRealVector ebeMultiply(RealVector v) {
        checkVectorDimensions(v.getDimension());
        OpenMapRealVector result = new OpenMapRealVector(v.getDimension());
        for (int i = 0; i < v.getDimension(); i++) {
            double value = 0.0;
            double thisValue = entries.containsKey(i) ? entries.get(i) : 0.0;
            double vValue = v.getEntry(i);
            if (Double.isNaN(thisValue) || Double.isNaN(vValue)) {
                value = Double.NaN;
            } else if (Double.isInfinite(thisValue) || Double.isInfinite(vValue)) {
                if (thisValue == 0.0 || vValue == 0.0) {
                    value = Double.NaN;
                } else {
                    value = thisValue * vValue;
                }
            } else {
                value = thisValue * vValue;
            }
            result.setEntry(i, value);
        }
        return result;
    }
