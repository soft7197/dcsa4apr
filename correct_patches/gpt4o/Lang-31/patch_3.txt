public static boolean containsAny(CharSequence cs, char[] searchChars) {
    if (isEmpty(cs) || ArrayUtils.isEmpty(searchChars)) {
        return false;
    }
    int csLength = cs.length();
    int searchLength = searchChars.length;
    outer:
    for (int i = 0; i < csLength; ) {
        int codePoint = Character.codePointAt(cs, i);
        for (int j = 0; j < searchLength; j++) {
            if (searchChars[j] == codePoint || (Character.isHighSurrogate(searchChars[j]) && j + 1 < searchLength && Character.isLowSurrogate(searchChars[j + 1]) && codePoint == Character.toCodePoint(searchChars[j], searchChars[j + 1]))) {
                return true;
            }
        }
        i += Character.charCount(codePoint);
    }
    return false;
}
