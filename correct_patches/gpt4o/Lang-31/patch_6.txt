public static boolean containsAny(CharSequence cs, char[] searchChars) {
    if (isEmpty(cs) || ArrayUtils.isEmpty(searchChars)) {
        return false;
    }
    int csLength = cs.length();
    int[] searchCodePoints = new int[searchChars.length];
    for (int k = 0; k < searchChars.length; k++) {
        searchCodePoints[k] = Character.codePointAt(searchChars, k);
    }
    for (int i = 0; i < csLength; i += Character.charCount(Character.codePointAt(cs, i))) {
        int ch = Character.codePointAt(cs, i);
        for (int searchCodePoint : searchCodePoints) {
            if (searchCodePoint == ch) {
                return true;
            }
        }
    }
    return false;
}
