public boolean isDirectory() {
    if (file != null) {
        return file.isDirectory();
    }

    if (linkFlag == LF_DIR) {
        return true;
    }

    if (linkFlag == LF_NORMAL || linkFlag == LF_DIR && file == null) {
        return getName().endsWith("/");
    }

    return false;
}
