public Attribute(String key, String val, Attributes parent) {
    Validate.notNull(key);
    String trimmedKey = key.trim();
    if (trimmedKey.isEmpty()) {
        throw new IllegalArgumentException("Key cannot be empty after trimming");
    }
    this.key = trimmedKey;
    this.val = val;
    this.parent = parent;
}
