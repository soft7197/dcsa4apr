private final static int _parseIndex(String str) {
    if (str == null) {
        return -1;
    }
    final int len = str.length();
    if (len == 0 || len > 10) {
        return -1;
    }
    for (int i = 0; i < len; ++i) {
        char c = str.charAt(i);
        if (!Character.isDigit(c)) {
            return -1;
        }
    }
    try {
        return Integer.parseInt(str);
    } catch (NumberFormatException e) {
        return -1;
    }
}
