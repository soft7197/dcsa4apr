public int parseInto(DateTimeParserBucket bucket, String text, int position) {
    if (text == null || text.isEmpty() || position < 0 || position >= text.length()) {
        return ~position;
    }
    String str = text.substring(position);
    String[] ALL_IDS_ARRAY = ALL_IDS.toArray(new String[0]);
    java.util.Arrays.sort(ALL_IDS_ARRAY, new java.util.Comparator<String>() {
        public int compare(String o1, String o2) {
            return o2.length() - o1.length();
        }
    });
    for (String id : ALL_IDS_ARRAY) {
        if (str.startsWith(id)) {
            bucket.setZone(DateTimeZone.forID(id));
            return position + id.length();
        }
    }
    return ~position;
}
