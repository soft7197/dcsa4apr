private String formatMethodCall() {
    StringBuilder sb = new StringBuilder(invocation.getMethod().getName());
    sb.append("(");
    for (int i = 0; i < invocation.getArguments().length; i++) {
        if (i > 0) sb.append(", ");
        sb.append(invocation.getArguments()[i]);
    }
    sb.append(")");
    return sb.toString();
}
