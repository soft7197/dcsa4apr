public void skipValue() throws IOException {
  if (peek() == JsonToken.NAME) {
    nextName();
    if (stackSize > 1) {
        pathNames[stackSize - 2] = "null";
    }
  } else {
    popStack();
    if (stackSize > 0) {
        pathNames[stackSize - 1] = "null";
        if (pathIndices != null && stackSize - 1 < pathIndices.length) {
            pathIndices[stackSize - 1]++;
        }
    }
  }
}
