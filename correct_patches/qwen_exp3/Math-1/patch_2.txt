private BigFraction(final double value, final double epsilon,
                        final int maxDenominator, int maxIterations)
        throws FractionConversionException {
        BigInteger overflow = BigInteger.valueOf(Integer.MAX_VALUE);
        double r0 = value;
        BigInteger a0 = BigInteger.valueOf((long) FastMath.floor(r0));
        if (a0.compareTo(overflow) > 0) {
            throw new FractionConversionException(value, a0.longValue(), 1L);
        }

        // check for (almost) integer arguments, which should not go
        // to iterations.
        if (FastMath.abs(a0.doubleValue() - value) < epsilon) {
            numerator = a0;
            denominator = BigInteger.ONE;
            return;
        }

        BigInteger p0 = BigInteger.ONE;
        BigInteger q0 = BigInteger.ZERO;
        BigInteger p1 = a0;
        BigInteger q1 = BigInteger.ONE;

        BigInteger p2 = BigInteger.ZERO;
        BigInteger q2 = BigInteger.ONE;

        int n = 0;
        boolean stop = false;
        do {
            ++n;
            final double r1 = 1.0 / (r0 - a0.doubleValue());
            final BigInteger a1 = BigInteger.valueOf((long) FastMath.floor(r1));
            p2 = a1.multiply(p1).add(p0);
            q2 = a1.multiply(q1).add(q0);

            final double convergent = p2.doubleValue() / q2.doubleValue();
            if ((n < maxIterations) &&
                (FastMath.abs(convergent - value) > epsilon) &&
                (q2.compareTo(BigInteger.valueOf(maxDenominator)) < 0)) {
                p0 = p1;
                p1 = p2;
                q0 = q1;
                q1 = q2;
                a0 = a1;
                r0 = r1;
            } else {
                stop = true;
            }
        } while (!stop);

        if (n >= maxIterations) {
            throw new FractionConversionException(value, maxIterations);
        }

        if (q2.compareTo(BigInteger.valueOf(maxDenominator)) < 0) {
            numerator   = p2;
            denominator = q2;
        } else {
            numerator   = p1;
            denominator = q1;
        }
    }
