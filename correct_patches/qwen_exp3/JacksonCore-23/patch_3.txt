public DefaultPrettyPrinter createInstance() {
    if (this.getClass() == DefaultPrettyPrinter.class) {
        return new DefaultPrettyPrinter(this);
    } else {
        throw new IllegalStateException("Subclass does not override createInstance");
    }
}
