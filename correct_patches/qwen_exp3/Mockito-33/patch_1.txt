public boolean hasSameMethod(Invocation candidate) {        
    if (candidate == null) return false;
    Method m1 = invocation.getMethod();
    Method m2 = candidate.getMethod();
    
    if (m1 == null || m2 == null) return false;
    
    return m1.getName().equals(m2.getName()) && sameParameterTypes(m1.getParameterTypes(), m2.getParameterTypes());
}

private boolean sameParameterTypes(Class<?>[] params1, Class<?>[] params2) {
    if (params1 == null || params2 == null) return false;
    if (params1.length != params2.length) return false;
    for (int i = 0; i < params1.length; i++) {
        if (!params1[i].equals(params2[i])) return false;
    }
    return true;
}
