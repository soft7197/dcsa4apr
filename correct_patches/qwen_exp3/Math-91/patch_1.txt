Method: Fraction.compareTo

public int compareTo(Fraction object) {
    if (object == null) {
        throw new NullPointerException("Object to compare cannot be null");
    }
    long thisNumerator = this.getNumerator() * (long) object.getDenominator();
    long otherNumerator = object.getNumerator() * (long) this.getDenominator();
    return Long.compare(thisNumerator, otherNumerator);
}
