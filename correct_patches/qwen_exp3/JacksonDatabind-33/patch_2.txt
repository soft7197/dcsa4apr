public PropertyName findNameForSerialization(Annotated a)
{
    String name = null;

    JsonGetter jg = _findAnnotation(a, JsonGetter.class);
    if (jg != null) {
        name = jg.value();
    } else {
        JsonProperty pann = _findAnnotation(a, JsonProperty.class);
        if (pann != null) {
            name = pann.value();
        } else if (_hasAnnotation(a, JsonSerialize.class)
                || _hasAnnotation(a, JsonView.class)
                || _hasAnnotation(a, JsonRawValue.class)) {
            name = "";
        }
    }

    JsonUnwrapped ju = _findAnnotation(a, JsonUnwrapped.class);
    if (ju != null && name == null) {
        // Handle unwrapped property, potentially using a default name or prefix
        name = ""; // This can be refined further based on specific requirements
    }

    if (name == null) {
        return null;
    }
    return PropertyName.construct(name);
}
