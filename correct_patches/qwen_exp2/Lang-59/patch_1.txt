Method: StrBuilder.appendFixedWidthPadRight

public StrBuilder appendFixedWidthPadRight(Object obj, int width, char padChar) {
    if (width > 0) {
        ensureCapacity(size + width);
        String str = (obj == null ? getNullText() : obj.toString());
        int strLen = str.length();
        int charsToWrite = Math.min(strLen, width);
        str.getChars(0, charsToWrite, buffer, size);
        int padLen = width - charsToWrite;
        for (int i = 0; i < padLen; i++) {
            buffer[size + charsToWrite + i] = padChar;
        }
        size += width;
    }
    return this;
}
