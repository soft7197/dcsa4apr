protected double getInitialDomain(double p) {
    double ret;
    double d = getDenominatorDegreesOfFreedom();
    if (d <= 2.0) {
        // For small degrees of freedom, use a small positive value to avoid division issues
        ret = 0.5;
    } else {
        // Use the mean for larger degrees of freedom
        ret = d / (d - 2.0);
    }
    return ret;
}
