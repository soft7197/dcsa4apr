public ObjectNode deserialize(JsonParser p, DeserializationContext ctxt) throws IOException
{
    if (p == null || ctxt == null) {
        throw new IllegalArgumentException("JsonParser and DeserializationContext cannot be null");
    }
    if (p.getCurrentToken() == JsonToken.START_OBJECT) {
        p.nextToken();
        return deserializeObject(p, ctxt, ctxt.getNodeFactory());
    }
    // Handle the case where the object is empty and parser is at END_OBJECT
    if (p.getCurrentToken() == JsonToken.END_OBJECT) {
        return ctxt.getNodeFactory().objectNode();
    }
    // If parser is at FIELD_NAME, it means we are inside an object and should continue deserializing
    if (p.getCurrentToken() == JsonToken.FIELD_NAME) {
        return deserializeObject(p, ctxt, ctxt.getNodeFactory());
    }
    // Throw an exception for unexpected tokens or malformed JSON
    throw ctxt.mappingException(ObjectNode.class);
}
