public int compareTo(Fraction object) {
    if (object == null) {
        throw new NullPointerException("Object to compare cannot be null");
    }
    if (denominator == 0 || object.denominator == 0) {
        throw new ArithmeticException("Cannot compare fractions with zero denominator");
    }
    long nOd = (long) numerator * object.denominator;
    long dOn = (long) object.numerator * denominator;
    return Long.compare(nOd, dOn);
}
