Method: Fraction.compareTo

public int compareTo(Fraction object) {
    long crossProductThis = (long) this.numerator * object.denominator;
    long crossProductOther = (long) object.numerator * this.denominator;
    return Long.compare(crossProductThis, crossProductOther);
}
