Method: Fraction.compareTo

public int compareTo(Fraction object) {
    if (object == null) {
        throw new NullPointerException("Cannot compare to null");
    }
    long numeratorDifference = (long)this.numerator * object.denominator - (long)object.numerator * this.denominator;
    return Long.signum(numeratorDifference);
}
