public int compareTo(Fraction object) {
    if (object == null) {
        throw new NullPointerException("Object to compare cannot be null");
    }
    if (denominator == 0 || object.denominator == 0) {
        throw new ArithmeticException("Comparison undefined for fractions with zero denominator");
    }
    long nOd = numerator * (long) object.denominator;
    long dOn = denominator * (long) object.numerator;
    return (nOd < dOn) ? -1 : ((nOd > dOn) ? +1 : 0);
}
