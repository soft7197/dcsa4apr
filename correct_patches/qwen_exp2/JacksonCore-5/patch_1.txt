private final static int _parseIndex(String str) {
    if (str == null || str.isEmpty() || str.length() > 10) {
        return -1;
    }
    try {
        long l = Long.parseLong(str);
        if (l > Integer.MAX_VALUE || l < 0) {
            return -1;
        }
        return (int) l;
    } catch (NumberFormatException e) {
        return -1;
    }
}
