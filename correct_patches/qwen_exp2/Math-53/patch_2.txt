public Complex add(Complex rhs)
    throws NullArgumentException {
    MathUtils.checkNotNull(rhs);
    if (Double.isNaN(real) || Double.isNaN(imaginary) || Double.isNaN(rhs.getReal()) || Double.isNaN(rhs.getImaginary())) {
        return NaN;
    }
    double realSum = real + rhs.getReal();
    double imagSum = imaginary + rhs.getImaginary();
    if (Double.isNaN(realSum) || Double.isNaN(imagSum)) {
        return NaN;
    }
    return createComplex(realSum, imagSum);
}
