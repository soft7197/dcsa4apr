public static Iterator collatedIterator(final Comparator comparator, final Collection iterators) {
    if (iterators == null) {
        throw new NullPointerException("Iterators collection must not be null");
    }
    if (iterators.contains(null)) {
        throw new NullPointerException("Iterators collection must not contain null values");
    }
    CollatingIterator collatingIterator = new CollatingIterator();
    collatingIterator.setComparator(comparator == null ? ComparatorUtils.naturalComparator() : comparator);
