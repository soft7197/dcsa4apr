public void addValue(Object v) {
    if (v == null) {
        throw new IllegalArgumentException("Object cannot be null");
    }
    if (!(v instanceof Comparable)) {
        throw new IllegalArgumentException("Object is not Comparable");
    }
    addValue((Comparable<?>) v);
}
