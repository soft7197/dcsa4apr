public void addValue(Object v) {
    if (v == null) {
        throw new IllegalArgumentException("Value cannot be null");
    }
    if (v instanceof Comparable) {
        addValue((Comparable<?>) v);
    } else {
        throw new IllegalArgumentException("Value is not Comparable");
    }
}
