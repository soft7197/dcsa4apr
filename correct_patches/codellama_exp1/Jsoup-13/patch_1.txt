public boolean hasAttr(String attributeKey) {
        Validate.notNull(attributeKey);

        if (attributeKey.startsWith("abs:")) {
            try {
                return absUrl(attributeKey.substring(4)).length() > 0;
            } catch (Exception e) {
                return false;
            }
        }

        return attributes.hasKey(attributeKey);
    }
