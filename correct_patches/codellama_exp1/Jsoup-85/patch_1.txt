public Attribute(String key, String val, Attributes parent) {
    Validate.notNull(key);
    this.key = key.trim();
    this.val = val;
    this.parent = parent;
    if (this.key.isEmpty()) {
        throw new IllegalArgumentException("Key cannot be empty");
    }
}
