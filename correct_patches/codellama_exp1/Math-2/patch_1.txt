Method: HypergeometricDistribution.getNumericalMean

public double getNumericalMean() {
    long sampleSize = (long) getSampleSize();
    long populationSize = (long) getPopulationSize();
    long numberOfSuccesses = (long) getNumberOfSuccesses();
    return (double) (sampleSize * numberOfSuccesses) / populationSize;
}
