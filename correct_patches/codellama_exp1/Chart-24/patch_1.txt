public Paint getPaint(double value) {
  double v = value;
  if (Math.abs(value - this.lowerBound) < 1e-6) v = this.lowerBound;
  if (Math.abs(value - this.upperBound) < 1e-6) v = this.upperBound;
  int g = Math.min(Math.max((int) ((v - this.lowerBound) / (this.upperBound - this.lowerBound) * 255.0), 0), 255);
  return new Color(g, g, g);
}
