public double getSumSquaredErrors() {
  if (sumXX == 0.0) return 0.0; // handle division by zero
  if (sumYY - sumXY * sumXY / sumXX < 0.0) return 0.0; // handle negative values
  return Math.max(0, sumYY - sumXY * sumXY / sumXX); // refine logic to handle edge cases
}
