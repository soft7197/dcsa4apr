public static boolean toBoolean(String str) {
    if (str == null) {
        return false;
    }
    switch (str.length()) {
        case 2: {
            return str.equalsIgnoreCase("on");
        }
        case 3: {
            return str.equalsIgnoreCase("yes");
        }
        case 4: {
            return str.equalsIgnoreCase("true");
        }
        default:
            return false;
    }
}
