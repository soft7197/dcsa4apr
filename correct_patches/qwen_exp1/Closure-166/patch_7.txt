public void matchConstraint(JSType constraint) {
  // We only want to match constraints on anonymous types.
  if (hasReferenceName() || constraint == null) {
    return;
  }

  // Handle the case where the constraint object is a union of record types.
  if (constraint.isUnionType()) {
    UnionType unionType = constraint.toMaybeUnionType();
    if (unionType != null) {
      for (JSType alternate : unionType.getAlternates()) {
        matchConstraint(alternate);
      }
    }
  } else if (constraint.isRecordType()) {
    matchRecordTypeConstraint(constraint.toObjectType());
  }
}
