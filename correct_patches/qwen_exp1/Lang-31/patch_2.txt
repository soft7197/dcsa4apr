public static boolean containsAny(CharSequence cs, char[] searchChars) {
	if (isEmpty(cs) || ArrayUtils.isEmpty(searchChars)) {
		return false;
	}
	int csLength = cs.length();
	int searchLength = searchChars.length;
	for (int i = 0; i < csLength; ) {
		char ch = cs.charAt(i);
		int codePoint;
		if (Character.isHighSurrogate(ch) && i + 1 < csLength && Character.isLowSurrogate(cs.charAt(i + 1))) {
			codePoint = Character.toCodePoint(ch, cs.charAt(i + 1));
			i += 2;
		} else {
			codePoint = ch;
			i++;
		}
		for (int j = 0; j < searchLength; ) {
			char searchCh = searchChars[j];
			int searchCodePoint;
			if (Character.isHighSurrogate(searchCh) && j + 1 < searchLength && Character.isLowSurrogate(searchChars[j + 1])) {
				searchCodePoint = Character.toCodePoint(searchCh, searchChars[j + 1]);
				j += 2;
			} else {
				searchCodePoint = searchCh;
				j++;
			}
			if (searchCodePoint == codePoint) {
				return true;
			}
		}
	}
	return false;
}
