static boolean preserveWhitespace(Node node) {
    // Traverse up the tree until we find an element that preserves whitespace or reach the root
    while (node != null && !(node instanceof Element)) {
        node = node.parent();
    }
    if (node == null) {
        return false;
    }
    Element el = (Element) node;
    while (el != null) {
        if (el.tag.preserveWhitespace()) {
            return true;
        }
        el = el.parent();
    }
    return false;
}
