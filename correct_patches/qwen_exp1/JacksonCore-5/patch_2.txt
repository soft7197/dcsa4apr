private final static int _parseIndex(String str) {
    if (str == null || str.isEmpty() || str.length() > 10) {
        return -1;
    }
    if (!str.matches("\\d+")) {
        return -1;
    }
    long l = NumberInput.parseLong(str);
    if (l > Integer.MAX_VALUE) {
        return -1;
    }
    return NumberInput.parseInt(str);
}
