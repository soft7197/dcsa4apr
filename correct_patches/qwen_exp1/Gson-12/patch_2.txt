public void skipValue() throws IOException {
  JsonToken token = peek();
  if (token == JsonToken.NAME) {
    nextName();
    if (stackSize > 1 && pathNames != null && stackSize - 2 < pathNames.length) {
        pathNames[stackSize - 2] = "null";
    }
  } else {
    popStack();
    if (stackSize > 0 && pathNames != null && stackSize - 1 < pathNames.length) {
        pathNames[stackSize - 1] = "null";
    }
  }
  if (stackSize > 0 && pathIndices != null && stackSize - 1 < pathIndices.length) {
      pathIndices[stackSize - 1]++;
  }
}
