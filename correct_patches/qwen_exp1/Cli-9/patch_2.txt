protected void checkRequiredOptions()
    throws MissingOptionException
{
    // Retrieve the list of required options that have not been processed
    List<Object> requiredOptions = getRequiredOptions();
    if (requiredOptions == null || requiredOptions.isEmpty()) {
        return; // No required options are missing, so do nothing
    }

    // Initialize a StringBuilder to construct the error message
    StringBuilder buff = new StringBuilder("Missing required option");
    buff.append(requiredOptions.size() == 1 ? "" : "s");
    buff.append(": ");

    // Loop through the required options and append them to the error message
    for (int i = 0; i < requiredOptions.size(); i++) {
        buff.append(requiredOptions.get(i));
        if (i < requiredOptions.size() - 1) {
            buff.append(", ");
        }
    }

    // Throw the MissingOptionException with the constructed error message
    throw new MissingOptionException(buff.toString());
}
