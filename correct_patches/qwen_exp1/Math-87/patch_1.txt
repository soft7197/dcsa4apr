private Integer getBasicRow(final int col) {
    Integer row = null;
    int nonZeroCount = 0;
    for (int i = getNumObjectiveFunctions(); i < getHeight(); i++) {
        double entry = getEntry(i, col);
        if (!MathUtils.equals(entry, 0.0, epsilon)) {
            if (MathUtils.equals(entry, 1.0, epsilon)) {
                nonZeroCount++;
                if (nonZeroCount > 1) {
                    return null;
                }
                row = i;
            } else {
                return null;
            }
        }
    }
    return row;
}
