static boolean isSimpleNumber(String s) {
  if (s == null || s.isEmpty()) {
    return false;
  }
  if (s.equals("0")) {
    return true;
  }
  int len = s.length();
  boolean hasLeadingZero = s.charAt(0) == '0';
  for (int index = 0; index < len; index++) {
    char c = s.charAt(index);
    if (!Character.isDigit(c)) {
      return false;
    }
    if (hasLeadingZero && index == 1) {
      return false;
    }
  }
  return true;
}
