private void copyAttributes(org.jsoup.nodes.Node source, Element el) {
    for (Attribute attribute : source.attributes()) {
        String key = attribute.getKey();
        if (key != null && !key.isEmpty()) {
            // valid xml attribute names are: ^[a-zA-Z_:][a-zA-Z0-9:._-]*
            key = key.replaceAll("^[^a-zA-Z_:]", ""); // Remove invalid starting character
            key = key.replaceAll("[^a-zA-Z0-9:._-]", ""); // Remove invalid subsequent characters
            if (!key.isEmpty()) { // Ensure the key is not empty after sanitization
                el.setAttribute(key, attribute.getValue());
            }
        }
    }
}
