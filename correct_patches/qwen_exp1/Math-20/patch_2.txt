public double[] repairAndDecode(final double[] x) {
    if (x == null) {
        throw new IllegalArgumentException("Input array cannot be null");
    }
    double[] decoded = decode(x);
    double[] lowerBounds = getLowerBound();
    double[] upperBounds = getUpperBound();
    if (lowerBounds == null || upperBounds == null) {
        throw new IllegalStateException("Bounds arrays cannot be null");
    }
    if (decoded.length != lowerBounds.length || decoded.length != upperBounds.length) {
        throw new IllegalArgumentException("Length of input array and bounds arrays must match");
    }
    for (int i = 0; i < decoded.length; i++) {
        decoded[i] = Math.max(lowerBounds[i], Math.min(decoded[i], upperBounds[i]));
    }
    return decoded;
}
