protected Object _deserializeFromEmptyString() throws IOException {
      switch (_kind) {
        case STD_URI:
          return URI.create("");
        case STD_LOCALE:
          return Locale.ROOT;
        default:
          return super._deserializeFromEmptyString();
      }
    }
