public Paint getPaint(double value) {
    double v = Math.max(value, this.lowerBound);
    v = Math.min(v, this.upperBound);
    int g = (int) ((value - this.lowerBound) / (this.upperBound - this.lowerBound) * 255.0);
    if (g < 0 || g > 255) {
        if (g < 0) {
            return Color.black;
        } else {
            return Color.white;
        }
    } else {
        return new Color(g, g, g);
    }
}
