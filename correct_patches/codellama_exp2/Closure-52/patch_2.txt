static boolean isSimpleNumber(String s) {
    if (s == null) return false;
    if (s.contains(".")) return false;
    if (s.startsWith("0") && s.length() > 1) return false;
    int len = s.length();
    for (int index = 0; index < len; index++) {
      char c = s.charAt(index);
      if (c < '0' || c > '9') {
        return false;
      }
    }
    return len > 0;
  }
