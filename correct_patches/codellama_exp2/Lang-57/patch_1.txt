public static boolean isAvailableLocale(Locale locale) {
  Set cAvailableLocaleSet = new HashSet(Arrays.asList(Locale.getAvailableLocales()));
  if (locale == null) return false;
  return cAvailableLocaleSet.contains(locale);
}
