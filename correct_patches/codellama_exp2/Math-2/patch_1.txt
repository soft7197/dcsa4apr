Method: HypergeometricDistribution.getNumericalMean

public double getNumericalMean(){
    int popSize = getPopulationSize();
    int sampleSize = getSampleSize();
    int successes = getNumberOfSuccesses();
    double mean = (double) successes * sampleSize / popSize;
    return mean;
}
