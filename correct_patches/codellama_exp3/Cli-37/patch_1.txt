private boolean isShortOption(String token) {
    if (token.startsWith("-")) {
      token = token.substring(1);
      int equalsIndex = token.indexOf('=');
      if (equalsIndex != -1) {
        token = token.substring(0, equalsIndex);
      }
    }
    return options.hasShortOption(token);
}
